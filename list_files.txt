Pour commencer, voici la fonction `main()`, la fonction `parse_inpupt()`, `parsing_mandelbrot()`, `parsing_julia()`, `parsing_sierpinski()`, et `parsing_sierpinski_depth()` :

```static int	parse_input(int argc, char **argv, t_fractal *fractal)
{
	if (argc == 2 && !ft_strcmp(argv[1], "mandelbrot"))
		parsing_mandelbrot(argv, fractal);
	else if (argc == 4 && !ft_strcmp(argv[1], "julia"))
		parsing_julia(argv, fractal);
	else if (argc == 2 && !ft_strcmp(argv[1], "sierpinski"))
		parsing_sierpinski(argv, fractal);
	else if (argc == 3 && !ft_strcmp(argv[1], "sierpinski"))
		parsing_sierpinski_depth(argv, fractal);
	else
		return (1);
	return (0);
}```

```int	main(int argc, char **argv)
{
	t_fractal	fractal;

	ft_str_tolower(argv[1]);
	if (parse_input(argc, argv, &fractal))
	{
		ft_putstr_fd("Usage :\n\n\t'./fractol mandelbrot'\n\
\t'./fractol julia <real> <imaginary>'\n\t'./fractol sierpinski'\n\t\
'./fractol sierpinski <depth>'\n\n", 1);
		return (1);
	}
	return (0);
}```
